**Goal:** Add a lightweight indexing and metadata layer on top of Robin's existing markdown memory files, enabling cross-file relationships, smart retrieval, and future archiving.

**Architecture:** Per-file sidecar indexes in `core/index/` store entry-level metadata (domains, tags, relationships, summaries) keyed by timestamp-based IDs embedded in source files. **Tech Stack:** Node.js 18+ (ES modules), `node:test` + `node:assert/strict`, Commander.js CLI, file-based markdown persistence.

// scripts/lib/index-utils.js

const seqTracker = { minute: '', count: 0 };

export function generateEntryId(sessionShort) {
  const now = new Date();
  const date = now.toISOString().slice(0, 10).replace(/-/g, '');
  const hours = String(now.getUTCHours()).padStart(2, '0');
  const mins = String(now.getUTCMinutes()).padStart(2, '0');
  const minute = `${date}-${hours}${mins}`;

  if (seqTracker.minute === minute) {
    seqTracker.count++;
  } else {
    seqTracker.minute = minute;
    seqTracker.count = 0;
  }

  const seq = String.fromCharCode(97 + seqTracker.count); // a=97
  return `${minute}-${sessionShort}${seq}`;
}

export function generateMigrationId(entryDate, seq, fallbackDate) {
  const dateStr = entryDate || fallbackDate || new Date().toISOString().slice(0, 10);
  const normalized = dateStr.replace(/-/g, '');
  const seqStr = String(seq).padStart(2, '0');
  return `${normalized}-0000-mig${seqStr}`;
}
```

export function parseAppendOnlyEntries(content) {
  const marker = '<!-- APPEND-ONLY below';
  const markerIdx = content.indexOf(marker);
  const body = markerIdx !== -1
    ? content.slice(content.indexOf('\n', markerIdx) + 1)
    : content;

  const blocks = body.split(/\n\n+/).filter(b => b.trim());
  const entries = [];

  for (const block of blocks) {
    if (block.includes('<!-- id:')) continue;

    const dateMatch = block.match(/\*\*(\d{4}-\d{2}-\d{2})\*\*/);
    const date = dateMatch ? dateMatch[1] : null;
    entries.push({ text: block.trim(), date });
  }

  return entries;
}

export function parseReferenceEntries(content) {
  const lines = content.split('\n');
  const entries = [];
  let currentSection = null;

  for (let i = 0; i < lines.length; i++) {
    const line = lines[i];

    if (line.startsWith('## ')) {
      currentSection = line.slice(3).trim();
      continue;
    }

    if (!currentSection) continue;
    if (!line.startsWith('- ')) continue;
    if (line.includes('<!-- id:')) continue;

    let text = line;
    const indented = [];
    let j = i + 1;
    while (j < lines.length && lines[j].match(/^\s+[-*]?\s/)) {
      indented.push(lines[j]);
      j++;
    }
    if (indented.length) text += '\n' + indented.join('\n');

    const boldMatch = line.match(/\*\*(.+?)\*\*/);
    const entity = boldMatch
      ? boldMatch[1].toLowerCase().replace(/\s+/g, '-').replace(/[^a-z0-9-]/g, '')
      : null;

    entries.push({ text: text.trim(), section: currentSection, entity, lineIndex: i });
  }

  return entries;
}

export function parseTaskEntries(content) {
  const lines = content.split('\n');
  const entries = [];
  let currentSection = null;

  for (let i = 0; i < lines.length; i++) {
    const line = lines[i];

    if (line.startsWith('## ')) {
      currentSection = line.slice(3).trim();
      continue;
    }

    if (!currentSection) continue;
    if (!line.match(/^- \[[ x]\] /)) continue;
    if (line.includes('<!-- id:')) continue;

    let text = line;
    let j = i + 1;
    while (j < lines.length && lines[j].match(/^\s+/)) {
      text += '\n' + lines[j];
      j++;
    }

    entries.push({ text: text.trim(), section: currentSection, lineIndex: i });
  }

  return entries;
}

export function injectIdIntoLine(line, id) {
  return `${line} <!-- id:${id} -->`;
}

export function injectIdBeforeBlock(block, id) {
  return `<!-- id:${id} -->\n${block}`;
}

export function injectIdsIntoFile(content, assignments) {
  const lines = content.split('\n');
  // Process in reverse order so line indexes remain valid
  const sorted = [...assignments].sort((a, b) => b.lineIndex - a.lineIndex);

  for (const { lineIndex, id, type } of sorted) {
    if (type === 'inline') {
      lines[lineIndex] = injectIdIntoLine(lines[lineIndex], id);
    } else if (type === 'block') {
      lines.splice(lineIndex, 0, `<!-- id:${id} -->`);
    }
  }

  return lines.join('\n');
}

export function generateSkeletonIndex(title, entries, level) {
  const lines = [`# Index: ${title}\n`];

  if (level === 'fact') {
    const bySec = groupBySection(entries);
    for (const [section, sectionEntries] of Object.entries(bySec)) {
      lines.push(`## ${section}\n`);
      for (const entry of sectionEntries) {
        lines.push(`- id: ${entry.id}`);
        if (entry.entity) lines.push(`  entity: ${entry.entity}`);
        lines.push(`  domains: []`);
        lines.push(`  related: []`);
        lines.push(`  enriched: false\n`);
      }
    }
  } else {
    lines.push(`## Entries\n`);
    for (const entry of entries) {
      lines.push(`- id: ${entry.id}`);
      if (entry.section) lines.push(`  section: ${entry.section}`);
      lines.push(`  domains: []`);
      lines.push(`  tags: []`);
      lines.push(`  related: []`);
      lines.push(`  summary: ~`);
      lines.push(`  enriched: false\n`);
    }
  }

  return lines.join('\n') + '\n';
}

export function generateManifest(files) {
  const now = new Date().toISOString();
  const lines = [
    '# Memory Manifest\n',
    'Generated by Dream. Do not edit manually.',
    `Last updated: ${now}\n`,
    '## Files\n',
  ];

  for (const f of files) {
    lines.push(`- file: ${f.name}`);
    lines.push(`  path: ${f.path}`);
    lines.push(`  index: ${f.indexPath}`);
    lines.push(`  type: ${f.type}`);
    lines.push(`  entries: ${f.entries}`);
    if (f.domains) lines.push(`  domains: [${f.domains.join(', ')}]`);
    if (f.sections) lines.push(`  sections: [${f.sections.join(', ')}]`);
    lines.push('');
  }

  return lines.join('\n') + '\n';
}

function groupBySection(entries) {
  const groups = {};
  for (const entry of entries) {
    const sec = entry.section || 'Uncategorized';
    if (!groups[sec]) groups[sec] = [];
    groups[sec].push(entry);
  }
  return groups;
}
```

// scripts/migrate-index.js
import { existsSync, readFileSync, writeFileSync, mkdirSync, cpSync, readdirSync, statSync } from 'fs';
import { join } from 'path';
import { findConfig } from './lib/find-config.js';
import { USER_DATA_FILES } from './lib/platforms.js';
import {
  generateMigrationId,
  parseAppendOnlyEntries,
  parseReferenceEntries,
  parseTaskEntries,
  injectIdsIntoFile,
  generateSkeletonIndex,
  generateManifest,
} from './lib/index-utils.js';

export async function migrateIndex(pkgRoot) {
  const configPath = findConfig();
  if (!configPath) {
    console.error("Error: No Robin workspace found. Run 'robin init' to create one.");
    process.exit(1);
  }
  const workspaceDir = join(configPath, '..');
  await migrateIndexInDir(workspaceDir);
}

export async function migrateIndexInDir(workspaceDir) {
  const configPath = join(workspaceDir, 'robin.config.json');
  const config = JSON.parse(readFileSync(configPath, 'utf-8'));

  console.log('Migrating to indexed memory (Phase A: structural)...');

  // Step 1: Backup (skip if already exists)
  const date = new Date().toISOString().slice(0, 10);
  const backupDir = join(workspaceDir, 'archive', `pre-index-${date}`);
  const existingBackups = existsSync(join(workspaceDir, 'archive'))
    ? readdirSync(join(workspaceDir, 'archive')).filter(e => e.startsWith('pre-index-'))
    : [];
  if (existingBackups.length === 0) {
    mkdirSync(backupDir, { recursive: true });
    for (const file of USER_DATA_FILES) {
      const src = join(workspaceDir, file);
      if (existsSync(src)) cpSync(src, join(backupDir, file));
    }
    console.log(`  Backup created at archive/pre-index-${date}/`);
  }

  // Step 2-4: Parse entries, assign IDs, inject into source files
  const indexDir = join(workspaceDir, 'index');
  mkdirSync(indexDir, { recursive: true });

  const manifestFiles = [];
  let globalSeq = 0;

  // Reference files (fact-level): profile, knowledge
  for (const fileName of ['profile.md', 'knowledge.md']) {
    const filePath = join(workspaceDir, fileName);
    if (!existsSync(filePath)) continue;

    const content = readFileSync(filePath, 'utf-8');
    const entries = parseReferenceEntries(content);

    const assignments = [];
    const indexEntries = [];

    for (const entry of entries) {
      globalSeq++;
      const fallbackDate = getFileMtime(filePath);
      const id = generateMigrationId(null, globalSeq, fallbackDate);
      assignments.push({ lineIndex: entry.lineIndex, id, type: 'inline' });
      indexEntries.push({ id, section: entry.section, entity: entry.entity, text: entry.text });
    }

    if (assignments.length > 0) {
      const updated = injectIdsIntoFile(content, assignments);
      writeFileSync(filePath, updated);
    }

    const baseName = fileName.replace('.md', '');
    const sections = extractSections(content);
    const indexContent = generateSkeletonIndex(capitalize(baseName), indexEntries, 'fact');
    writeFileSync(join(indexDir, `${baseName}.idx.md`), indexContent);

    manifestFiles.push({
      name: baseName, path: fileName, indexPath: `index/${baseName}.idx.md`,
      type: 'reference', entries: indexEntries.length, sections,
    });
  }

  // Task file
  {
    const filePath = join(workspaceDir, 'tasks.md');
    if (existsSync(filePath)) {
      const content = readFileSync(filePath, 'utf-8');
      const entries = parseTaskEntries(content);

      const assignments = [];
      const indexEntries = [];

      for (const entry of entries) {
        globalSeq++;
        const id = generateMigrationId(null, globalSeq, getFileMtime(filePath));
        assignments.push({ lineIndex: entry.lineIndex, id, type: 'inline' });
        indexEntries.push({ id, section: entry.section, text: entry.text });
      }

      if (assignments.length > 0) {
        const updated = injectIdsIntoFile(content, assignments);
        writeFileSync(filePath, updated);
      }

      const sections = extractSections(content);
      const indexContent = generateSkeletonIndex('Tasks', indexEntries, 'entry');
      writeFileSync(join(indexDir, 'tasks.idx.md'), indexContent);

      manifestFiles.push({
        name: 'tasks', path: 'tasks.md', indexPath: 'index/tasks.idx.md',
        type: 'mixed', entries: indexEntries.length, sections,
      });
    }
  }

  // Append-only files: journal, decisions, inbox
  for (const fileName of ['journal.md', 'decisions.md', 'inbox.md']) {
    const filePath = join(workspaceDir, fileName);
    if (!existsSync(filePath)) continue;

    const content = readFileSync(filePath, 'utf-8');
    const entries = parseAppendOnlyEntries(content);
    const lines = content.split('\n');

    const assignments = [];
    const indexEntries = [];

    for (const entry of entries) {
      globalSeq++;
      const id = generateMigrationId(entry.date, globalSeq, getFileMtime(filePath));

      // Find the line index for this entry in the original file
      const entryFirstLine = entry.text.split('\n')[0];
      const lineIndex = lines.findIndex(l => l.trim() === entryFirstLine.trim());
      if (lineIndex === -1) continue;

      assignments.push({ lineIndex, id, type: 'block' });
      indexEntries.push({ id, text: entry.text, date: entry.date });
    }

    if (assignments.length > 0) {
      const updated = injectIdsIntoFile(content, assignments);
      writeFileSync(filePath, updated);
    }

    const baseName = fileName.replace('.md', '');
    const indexContent = generateSkeletonIndex(capitalize(baseName), indexEntries, 'entry');
    writeFileSync(join(indexDir, `${baseName}.idx.md`), indexContent);

    manifestFiles.push({
      name: baseName, path: fileName, indexPath: `index/${baseName}.idx.md`,
      type: 'append-only', entries: indexEntries.length,
    });
  }

  // Self-improvement (entry-level per subsection)
  {
    const filePath = join(workspaceDir, 'self-improvement.md');
    if (existsSync(filePath)) {
      const content = readFileSync(filePath, 'utf-8');
      // For migration, just create an empty skeleton — the subsections
      // have specific formats and most start empty in a fresh workspace
      const indexContent = generateSkeletonIndex('Self-Improvement', [], 'entry');
      writeFileSync(join(indexDir, 'self-improvement.idx.md'), indexContent);

      manifestFiles.push({
        name: 'self-improvement', path: 'self-improvement.md',
        indexPath: 'index/self-improvement.idx.md',
        type: 'mixed', entries: 0,
      });
    }
  }

  // Trips index (scan for trips/ directory)
  {
    const tripsDir = join(workspaceDir, 'trips');
    const tripEntries = [];
    if (existsSync(tripsDir)) {
      const tripFiles = readdirSync(tripsDir).filter(f => f.endsWith('.md'));
      for (const f of tripFiles) {
        globalSeq++;
        const slug = f.replace('.md', '');
        const id = generateMigrationId(null, globalSeq, getFileMtime(join(tripsDir, f)));
        tripEntries.push({ id, entity: slug, section: 'Trips', text: slug });
      }
    }
    const indexContent = generateSkeletonIndex('Trips', tripEntries, 'fact');
    writeFileSync(join(indexDir, 'trips.idx.md'), indexContent);
  }

  // Step 6: Generate manifest
  const manifestContent = generateManifest(manifestFiles);
  writeFileSync(join(workspaceDir, 'manifest.md'), manifestContent);

  // Step 7: Update config
  config.version = '2.1.0';
  config.indexing = {
    status: 'structural',
    migrated_at: new Date().toISOString(),
  };
  writeFileSync(configPath, JSON.stringify(config, null, 2) + '\n');

  console.log(`Migration complete (Phase A). ${globalSeq} entries indexed.`);
  console.log('Phase B (semantic enrichment) will run on your next Robin session.');
}

function getFileMtime(filePath) {
  try {
    const stat = statSync(filePath);
    return stat.mtime.toISOString().slice(0, 10);
  } catch {
    return new Date().toISOString().slice(0, 10);
  }
}

function extractSections(content) {
  const matches = content.match(/^## (.+)$/gm);
  return matches ? matches.map(m => m.slice(3).trim()) : [];
}

function capitalize(str) {
  return str.charAt(0).toUpperCase() + str.slice(1);
}

      const profile = readFileSync(profilePath, 'utf-8');
      const peopleIdx = profile.indexOf('## People');
      if (peopleIdx !== -1) {
        const insertAt = profile.indexOf('\n', peopleIdx) + 1;
        const updated = profile.slice(0, insertAt) + '\n- **Dr. Test** — Test doctor\n' + profile.slice(insertAt);
        writeFileSync(profilePath, updated);
      }

      // Run migration
      await migrateIndexInDir(lifecycleDir);
