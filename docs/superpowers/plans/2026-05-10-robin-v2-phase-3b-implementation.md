import { spawnSync } from 'node:child_process';
import { readdir, stat, unlink } from 'node:fs/promises';
import { join } from 'node:path';

function timestamp() {
  const d = new Date();
  const pad = (n) => String(n).padStart(2, '0');
  return `${d.getUTCFullYear()}${pad(d.getUTCMonth() + 1)}${pad(d.getUTCDate())}-${pad(d.getUTCHours())}${pad(d.getUTCMinutes())}${pad(d.getUTCSeconds())}`;
}

async function pruneOldBackups(backupDir, maxAgeMs) {
  const now = Date.now();
  let entries;
  try { entries = await readdir(backupDir); } catch (e) { if (e.code === 'ENOENT') return 0; throw e; }
  let pruned = 0;
  for (const f of entries) {
    if (!f.endsWith('.tar')) continue;
    const full = join(backupDir, f);
    const st = await stat(full);
    if (!st.isFile()) continue;
    if (now - st.mtimeMs > maxAgeMs) {
      await unlink(full);
      pruned++;
    }
  }
  return pruned;
}

export async function snapshot(srcDir, backupDir) {
  let entries;
  try { entries = await readdir(srcDir); } catch (e) { if (e.code === 'ENOENT') return null; throw e; }
  if (entries.length === 0) return null;

  const retentionDays = Number(process.env.ROBIN_BACKUP_RETENTION_DAYS ?? 30);
  if (retentionDays > 0) {
    await pruneOldBackups(backupDir, retentionDays * 86400_000);
  }

  const archive = join(backupDir, `${timestamp()}.tar`);
  const result = spawnSync('tar', ['-cf', archive, '-C', srcDir, '.'], { stdio: 'pipe' });
  if (result.status !== 0) {
    throw new Error(`tar failed (status=${result.status}): ${result.stderr?.toString()}`);
  }
  return archive;
}
```

import { sha256 } from '../embed/hash.js';

/**
 * Stable v1-id-based dedup hash. Input v1_id already includes the table
 * prefix (e.g., 'capture:abc123'). The 'v1:' prefix namespaces the hash
 * so future migration sources don't collide.
 */
export function sourceHash(v1_id) {
  return sha256(`v1:${v1_id}`);
}

export function buildFromV1({ v1_table, v1_id, migrated_at }) {
  return {
    v1_table,
    v1_id,
    source_hash: sourceHash(v1_id),
    migrated_at: migrated_at ?? new Date().toISOString(),
  };
}
```

import { surql } from 'surrealdb';

const ID = "type::record('runtime', 'migration_failures')";

async function readEntries(db) {
  const [rows] = await db.query(`SELECT * FROM ${ID}`).collect();
  return rows[0]?.value?.entries ?? [];
}

export async function recordFailure(db, { v1_table, v1_id, error_message, phase = null }) {
  const entries = await readEntries(db);
  entries.push({
    v1_table,
    v1_id,
    phase,
    error_message: String(error_message),
    occurred_at: new Date().toISOString(),
  });
  await db.query(surql`UPSERT type::record('runtime', 'migration_failures') SET value = ${{ entries }}`).collect();
}

export async function listFailures(db, { phase = null } = {}) {
  const entries = await readEntries(db);
  if (phase) return entries.filter((e) => e.phase === phase || e.v1_table === phase);
  return entries;
}

export async function clearFailures(db) {
  await db.query(surql`UPSERT type::record('runtime', 'migration_failures') SET value = ${{ entries: [] }}`).collect();
}

import { surql } from 'surrealdb';

const ID = "type::record('runtime', 'migration_id_map')";

export function createResolver(db) {
  const maps = { entity: new Map(), episode: new Map(), capture: new Map() };

  return {
    set(kind, v1_id, v2_id) {
      if (!maps[kind]) throw new Error(`unknown resolver kind: ${kind}`);
      maps[kind].set(v1_id, v2_id);
    },
    get(kind, v1_id) {
      return maps[kind]?.get(v1_id) ?? null;
    },
    has(kind, v1_id) {
      return maps[kind]?.has(v1_id) ?? false;
    },
    sizes() {
      return Object.fromEntries(Object.entries(maps).map(([k, m]) => [k, m.size]));
    },
    async persist() {
      const value = {
        entity:  Object.fromEntries(maps.entity),
        episode: Object.fromEntries(maps.episode),
        capture: Object.fromEntries(maps.capture),
      };
      await db.query(surql`UPSERT type::record('runtime', 'migration_id_map') SET value = ${value}`).collect();
    },
    async load() {
      const [rows] = await db.query(`SELECT * FROM ${ID}`).collect();
      const value = rows[0]?.value ?? {};
      for (const kind of ['entity', 'episode', 'capture']) {
        maps[kind] = new Map(Object.entries(value[kind] ?? {}));
      }
    },
  };
}

import { connect, close } from '../db/client.js';
import { surql } from 'surrealdb';
import { existsSync } from 'node:fs';
import { join } from 'node:path';

/**
 * Open v1's rocksdb at <packageRoot>/user-data/runtime/db/data, read-only by
 * convention. v1 must be stopped — RocksDB is single-writer; a running v1
 * daemon will fail this open with a lock error, which we surface verbatim.
 */
export async function openV1(packageRoot) {
  const dataDir = packageRoot.endsWith('/data')
    ? packageRoot
    : join(packageRoot, 'user-data/runtime/db/data');
  if (!existsSync(dataDir)) throw new Error(`v1 data dir not found: ${dataDir}`);
  let db;
  try {
    db = await connect({ engine: `rocksdb://${dataDir}` });
  } catch (e) {
    if (/lock|LOCK|locked/i.test(String(e?.message))) {
      throw new Error(`v1 appears to still be running — stop it first (rocksdb lock at ${dataDir})`);
    }
    throw e;
  }
  await db.use({ namespace: 'robin', database: 'main' });
  return {
    raw: db,
    async query(sql, ...args) {
      return db.query(sql, ...args);
    },
    async close() { await close(db); },
  };
}

export async function listTableCount(v1, table) {
  const safe = String(table).replace(/[^a-z_0-9]/gi, '');
  const [rows] = await v1.query(`SELECT count() AS n FROM ${safe} GROUP ALL`).collect();
  return rows[0]?.n ?? 0;
}

/**
 * Async-iterator over a v1 table in deterministic id order. Pages by id
 * cursor (id > $last) which is monotonic for SurrealDB ULIDs.
 */
export async function* scanTable(v1, table, { batch = 200, startAfter = null } = {}) {
  const safe = String(table).replace(/[^a-z_0-9]/gi, '');
  let cursor = startAfter;
  for (;;) {
    const sql = cursor === null
      ? `SELECT * FROM ${safe} ORDER BY id LIMIT ${Number(batch)}`
      : `SELECT * FROM ${safe} WHERE id > ${cursor} ORDER BY id LIMIT ${Number(batch)}`;
    const [rows] = await v1.query(sql).collect();
    if (!rows || rows.length === 0) return;
    yield rows;
    cursor = rows[rows.length - 1].id;
    if (rows.length < batch) return;
  }
}

      continue;

  }

      if (!v2In || !v2Out) {
        // Endpoint entity wasn't migrated (failed or not present); record + skip.
        await recordFailure(v2db, {
          v1_table: 'participates_in', v1_id: String(v1Row.id),
          error_message: `endpoint not in resolver (in=${v1Row.in} out=${v1Row.out})`,
          phase: 'edges',
        });
        skipped++; continue;
      }
      const hash = sourceHash(String(v1Row.id));
      // Dedup: query the participates_in edge table for matching from_v1 hash
      const [existing] = await v2db
        .query(surql`SELECT id FROM participates_in WHERE meta.from_v1.source_hash = ${hash} LIMIT 1`)
        .collect();
      if (existing[0]?.id) { dup++; continue; }
      try {
        const payload = buildParticipatesInPayload(v1Row);
        await v2db
          .query(surql`RELATE type::thing(${v2In})->participates_in->type::thing(${v2Out}) SET meta = ${payload.meta}`)
          .collect();
        imported++;
      } catch (e) {
        await recordFailure(v2db, {
          v1_table: 'participates_in', v1_id: String(v1Row.id),
          error_message: e.message, phase: 'edges',
        });
      }
      lastId = v1Row.id;
    }
    progress.advance({ phase: 'edges', last_v1_id: String(lastId), imported, dup, skipped });
  }
  return { imported, dup, skipped };
}

export function buildLossyEvent(table, v1Row, ctx = {}) {
  const def = LOSSY_TABLES.find((t) => t.table === table);
  if (!def) throw new Error(`unknown lossy table: ${table}`);
  const content = def.isEdge ? edgeSummary(table, v1Row) : (def.summarize?.(v1Row) ?? '');
  const safeContent = String(content).length === 0 ? `(v1 ${table} ${v1Row.id})` : String(content);
  const v1_payload = def.isEdge ? buildEdgePayload(table, v1Row, ctx) : (def.payload?.(v1Row, ctx) ?? {});
  return {
    content: safeContent,
    source: 'migration',
    content_hash: sha256(safeContent),
    ts: v1Row.ts ?? v1Row.created ?? v1Row.date ?? new Date().toISOString(),
    external_id: `v1:${v1Row.id}`,
    trust: 'trusted',
    embedding: null,
    meta: {
      kind: `v1_${table}`,
      v1_payload,
      from_v1: buildFromV1({ v1_table: table, v1_id: String(v1Row.id) }),
    },
  };
}

      for await (const batch of scanTable(v1, def.table, { batch: 200, startAfter: lastId })) {
        any = true;

          const hash = sourceHash(String(v1Row.id));
          const [existing] = await v2db
            .query(surql`SELECT id FROM events WHERE meta.from_v1.source_hash = ${hash} LIMIT 1`)
            .collect();

          try {
            const ev = buildLossyEvent(def.table, v1Row, { resolver });

          } catch (e) {

              error_message: e.message, phase: `lossy:${def.table}`,

        progress.advance({ phase: `lossy:${def.table}`, last_v1_id: String(lastId), imported, dup });

      // Table may not exist in this v1 instance (e.g. some custom v1 build) — continue.
      if (!/(?:not found|does not exist)/i.test(String(e?.message))) throw e;
    }
    stats[def.table] = { imported, dup, present: any };

import { surql } from 'surrealdb';
import { openV1 } from './v1-client.js';
import { createResolver } from './resolver.js';
import { runEntityPhase } from './phases/entity.js';
import { runEpisodePhase } from './phases/episode.js';
import { runCapturePhase } from './phases/capture.js';
import { runEdgesPhase } from './phases/edges.js';
import { runLossyPhase } from './phases/lossy.js';

const PROGRESS_ID = "type::record('runtime', 'migration_progress')";

async function readProgress(db) {
  const [rows] = await db.query(`SELECT * FROM ${PROGRESS_ID}`).collect();
  return rows[0]?.value?.v1_to_v2 ?? null;
}

async function writeProgress(db, value) {
  await db.query(surql`UPSERT type::record('runtime', 'migration_progress') SET value = ${{ v1_to_v2: value }}`).collect();
}

function makeProgress(initial, db) {
  const state = initial ?? {
    started_at: new Date().toISOString(),
    completed_phases: [],
    current_phase: null,
    cursor: {},
    counts: {},
  };
  return {
    state,
    advance({ phase, last_v1_id, imported = 0, dup = 0, skipped = 0 }) {
      state.current_phase = phase;
      state.cursor = { ...state.cursor, [phase]: { last_v1_id }};
      state.counts[phase] = state.counts[phase] ?? { imported: 0, dup: 0, skipped: 0 };
      state.counts[phase].imported += imported;
      state.counts[phase].dup += dup;
      state.counts[phase].skipped += skipped;
      // best-effort persist
      writeProgress(db, state).catch(() => {});
    },
    completePhase(phase) {
      if (!state.completed_phases.includes(phase)) state.completed_phases.push(phase);
      state.current_phase = null;
      return writeProgress(db, state);
    },
    get cursor() { return state.cursor; },
  };
}

const ALL_PHASES = ['entity', 'episode', 'capture', 'edges', 'lossy'];

export async function runMigration({ sourcePath, v2db, embedder, log = console.log, only = null }) {
  const v1 = await openV1(sourcePath);
  try {
    const resolver = createResolver(v2db);
    await resolver.load();

    const initial = await readProgress(v2db);
    const progress = makeProgress(initial, v2db);

    const phases = only ? [only] : ALL_PHASES;
    const out = { phases: {} };

    for (const ph of phases) {
      if (progress.state.completed_phases.includes(ph)) {
        log(`[migrate-v1] ${ph} already complete; skipping`);
        out.phases[ph] = { imported: 0, dup: 0, alreadyDone: true };
        continue;
      }
      log(`[migrate-v1] starting phase: ${ph}`);
      const ctx = { v1, v2db, resolver, embedder, progress };
      let r;
      if (ph === 'entity')   r = await runEntityPhase(ctx);
      else if (ph === 'episode')  r = await runEpisodePhase(ctx);
      else if (ph === 'capture')  r = await runCapturePhase(ctx);
      else if (ph === 'edges')    r = await runEdgesPhase(ctx);
      else if (ph === 'lossy')    r = await runLossyPhase(ctx);
      else throw new Error(`unknown phase: ${ph}`);
      out.phases[ph] = r;
      await resolver.persist();
      await progress.completePhase(ph);
      log(`[migrate-v1] phase ${ph} done: ${JSON.stringify(r)}`);
    }
    return out;
  } finally {
    await v1.close();
  }
}

import { surql } from 'surrealdb';

export async function embedBackfillTick({ db, embedder, batch = 64, log = () => {} }) {
  const [rows] = await db
    .query(surql`
      SELECT id, content FROM events
      WHERE embedding IS NONE AND meta.embed_failed IS NOT true
      ORDER BY ts ASC
      LIMIT ${batch}
    `)
    .collect();
  if (!rows || rows.length === 0) return { embedded: 0, failed: 0 };
  const start = Date.now();
  let embedded = 0, failed = 0;

  // Try whole-batch first
  let vecs;
  try {
    vecs = await embedder.embedBatch(rows.map((r) => r.content));
  } catch {
    vecs = null;
  }

  for (let i = 0; i < rows.length; i++) {
    const r = rows[i];
    try {
      const v = vecs ? vecs[i] : await embedder.embed(r.content);
      const arr = Array.from(v);
      await db.query(surql`UPDATE ${r.id} SET embedding = ${arr}, embedded_at = time::now()`).collect();
      embedded++;
    } catch (e) {
      await db.query(surql`UPDATE ${r.id} SET meta.embed_failed = true`).collect();
      failed++;
    }
  }
  log(`[embed_backfill] embedded ${embedded}, failed ${failed} (latency ${Date.now() - start}ms)`);
  return { embedded, failed };
}

const ACTIVITY_DAYS = 30;

function describeEvent(ev) {
  const t = ev.type, p = ev.payload ?? {};
  if (t === 'PushEvent') return `pushed ${p.commits?.length ?? 0} commit(s)`;
  if (t === 'PullRequestEvent') return `${p.action} PR #${p.number}: ${p.pull_request?.title ?? ''}`;
  if (t === 'IssuesEvent') return `${p.action} issue #${p.issue?.number}: ${p.issue?.title ?? ''}`;
  if (t === 'ReleaseEvent') return `${p.action} release ${p.release?.tag_name ?? ''}`;
  if (t === 'WatchEvent') return `starred ${ev.repo?.name ?? ''}`;
  return t;
}

async function gh(ctx, url) {
  const res = await ctx.fetchFn(url, {
    headers: { Authorization: `Bearer ${ctx.secrets.GITHUB_PAT}`, Accept: 'application/vnd.github+json' },
    signal: ctx.signal,
  });
  if (!res.ok) {
    const err = new Error(`github ${url}: ${res.status}`);
    err.status = res.status;
    throw err;
  }
  return await res.json();
}

export async function sync(ctx) {
  const me = await gh(ctx, 'https://api.github.com/user');
  const since = new Date(Date.now() - ACTIVITY_DAYS * 86400_000).toISOString();
  const events = [];

  const userEvents = await gh(ctx, `https://api.github.com/users/${me.login}/events?per_page=100`);
  for (const e of userEvents) {
    if (e.created_at < since) continue;
    events.push({
      external_id: `github:event:${e.id}`,
      content: `${e.repo?.name ?? ''}: ${e.type} — ${describeEvent(e)}`,
      ts: e.created_at,
      meta: { kind: 'github_activity', repo: e.repo?.name, type: e.type, raw: e },
    });
  }

  let notifs = [];
  try {
    notifs = await gh(ctx, 'https://api.github.com/notifications?all=true');
  } catch (e) {
    if (e.status === 403) ctx.log('/notifications 403 (fine-grained PAT?) — skipping');
    else throw e;
  }
  for (const n of notifs) {
    events.push({
      external_id: `github:notif:${n.id}`,
      content: `${n.repository?.full_name ?? ''}: [${n.reason}] ${n.subject?.type ?? ''} — ${n.subject?.title ?? ''}`,
      ts: n.updated_at,
      meta: { kind: 'github_notif', repo: n.repository?.full_name, reason: n.reason, unread: n.unread },
    });
  }

  // Releases from starred — capped at 50 repos to keep first-sync bounded
  let starred = [];
  try { starred = await gh(ctx, 'https://api.github.com/user/starred?per_page=50'); } catch {}
  for (const repo of starred.slice(0, 50)) {
    try {
      const rels = await gh(ctx, `https://api.github.com/repos/${repo.full_name}/releases?per_page=10`);
      for (const r of rels ?? []) {
        const t = r.published_at ?? r.created_at;
        if (!t || t < since) continue;
        events.push({
          external_id: `github:release:${repo.full_name}:${r.tag_name}`,
          content: `${repo.full_name} released ${r.tag_name}: ${r.name ?? ''}`,
          ts: t,
          meta: { kind: 'github_release', repo: repo.full_name, tag: r.tag_name },
        });
      }
    } catch { /* skip 404/etc */ }
  }

  await ctx.capture(events);
  return { count: events.length, cursor: { last_synced_at: new Date().toISOString() } };
}

const WINDOWS = ['short_term', 'medium_term', 'long_term'];

function monthBucket(d = new Date()) {
  return `${d.getUTCFullYear()}-${String(d.getUTCMonth() + 1).padStart(2, '0')}`;
}

export async function sync(ctx) {
  const accessToken = await getAccessToken({
    provider: 'spotify',
    secrets: ctx.secrets,
    fetchFn: ctx.fetchFn,
    saveSecret: ctx.saveSecret,
  });
  const client = ctx._client ?? createSpotifyClient(accessToken, { fetchFn: ctx.fetchFn, signal: ctx.signal });
  const me = await client.me();
  ctx.log(`account: ${me.display_name ?? me.id}`);

  const events = [];
  const lastPlayedAt = ctx.cursor?.last_played_at ?? null;

  const rp = await client.recentlyPlayed({ limit: 50 });
  const items = rp.items ?? [];
  let gap = false;
  if (lastPlayedAt && items.length === 50) {
    const oldest = items.reduce((a, b) => (a.played_at < b.played_at ? a : b))?.played_at;
    if (oldest && oldest > lastPlayedAt) gap = true;
  }
  for (const it of items) {
    const t = it.track ?? {};
    const artists = (t.artists ?? []).map((a) => a.name).join(', ');
    events.push({
      external_id: `spotify:played:${it.played_at}`,
      content: `played ${t.name ?? '?'} by ${artists} — ${t.album?.name ?? ''}`,
      ts: it.played_at,
      meta: { kind: 'spotify_played', track_id: t.id, artists, album: t.album?.name },
    });
  }
  if (gap) ctx.log('[spotify] gap detected: >50 plays since last sync; consider tighter cadence');

  const bucket = monthBucket();
  for (const w of WINDOWS) {
    const tracks = (await client.topItems('tracks', { time_range: w, limit: 50 })).items ?? [];
    for (const tr of tracks) {
      const artists = (tr.artists ?? []).map((a) => a.name).join(', ');
      events.push({
        external_id: `spotify:top_track:${w}:${bucket}:${tr.id}`,
        content: `top track (${w}) ${bucket}: ${tr.name} by ${artists}`,
        ts: new Date().toISOString(),
        meta: { kind: 'spotify_top_track', window: w, month: bucket, track_id: tr.id, artists },
      });
    }
    const artists = (await client.topItems('artists', { time_range: w, limit: 50 })).items ?? [];
    for (const ar of artists) {
      events.push({
        external_id: `spotify:top_artist:${w}:${bucket}:${ar.id}`,

      });
    }
  }

  await ctx.capture(events);
  const newest = items[0]?.played_at ?? lastPlayedAt;
  return {
    count: events.length,

      last_played_at: newest,
      last_top_refresh_at: new Date().toISOString(),

const REQUIRED = ['Date', 'Name', 'Year', 'Letterboxd URI', 'Watched Date'];

function splitCsvLine(line) {
  // Letterboxd export uses quoted strings only when needed; basic RFC-4180.
  const out = [];
  let cur = '', q = false;
  for (let i = 0; i < line.length; i++) {
    const c = line[i];
    if (c === '"' && line[i + 1] === '"') { cur += '"'; i++; }
    else if (c === '"') q = !q;
    else if (c === ',' && !q) { out.push(cur); cur = ''; }
    else cur += c;
  }
  out.push(cur);
  return out;
}

function slugFromUri(uri) {
  const m = String(uri).match(/film\/([^/]+)\/?/);
  return m ? m[1] : '';
}

export function isDiaryCsv(headerLine) {
  const cols = splitCsvLine(headerLine).map((c) => c.trim());
  return REQUIRED.every((r) => cols.some((c) => c.toLowerCase() === r.toLowerCase()));
}

export function parseDiaryCsv(text) {
  const lines = text.split(/\r?\n/).filter((l) => l.trim().length > 0);
  if (lines.length === 0) return [];
  if (!isDiaryCsv(lines[0])) {
    const err = new Error('not a Letterboxd diary CSV (missing required headers)');
    err.code = 'NOT_DIARY';
    throw err;
  }
  const headers = splitCsvLine(lines[0]).map((c) => c.trim());
  const idx = (name) => headers.findIndex((h) => h.toLowerCase() === name.toLowerCase());
  const I = {
    date: idx('Date'), name: idx('Name'), year: idx('Year'),
    uri: idx('Letterboxd URI'), rating: idx('Rating'),
    rewatch: idx('Rewatch'), tags: idx('Tags'), watched: idx('Watched Date'),
  };
  const rows = [];
  for (let i = 1; i < lines.length; i++) {
    const c = splitCsvLine(lines[i]);
    rows.push({
      date: c[I.date],
      name: c[I.name],
      year: c[I.year],
      uri: c[I.uri],
      slug: slugFromUri(c[I.uri]),
      rating: c[I.rating] ? Number(c[I.rating]) : null,
      rewatch: /yes/i.test(c[I.rewatch] ?? ''),
      tags: (c[I.tags] ?? '').split(',').map((s) => s.trim()).filter(Boolean),
      watched_date: c[I.watched],
    });
  }
  return rows;
}

import { join } from 'node:path';
import { paths } from '../../runtime/home.js';
import { isDiaryCsv, parseDiaryCsv } from './csv.js';

export async function sync(ctx) {
  const upload = join(paths().home, 'upload');
  const processed = join(upload, 'processed');
  if (!existsSync(upload)) return { count: 0 };
  mkdirSync(processed, { recursive: true });

  const candidates = readdirSync(upload).filter((f) => /^letterboxd-.*\.csv$/i.test(f));
  if (candidates.length === 0) return { count: 0 };

  const events = [];
  for (const f of candidates) {
    const full = join(upload, f);
    let text;
    try { text = readFileSync(full, 'utf-8'); }
    catch (e) { ctx.log(`letterboxd: read failed for ${f}: ${e.message}`); continue; }
    const headerLine = text.split(/\r?\n/)[0] ?? '';
    if (!isDiaryCsv(headerLine)) {
      ctx.log(`letterboxd: ${f} is not a diary CSV — moving aside`);
      const dest = join(processed, `${f}.unrecognized`);
      try { renameSync(full, dest); } catch {}
      try { writeFileSync(`${dest}.error.txt`, 'expected header columns: Date, Name, Year, Letterboxd URI, Watched Date\n'); } catch {}
      continue;
    }
    let rows;
    try { rows = parseDiaryCsv(text); }
    catch (e) {
      ctx.log(`letterboxd: parse failed for ${f}: ${e.message}`);
      try { writeFileSync(`${full}.error.txt`, e.message); } catch {}
      continue;
    }
    for (const r of rows) {
      const star = r.rating !== null ? `${r.rating}★` : '';
      const rewatchMarker = r.rewatch ? ' (rewatch)' : '';
      events.push({
        external_id: `letterboxd:diary:${r.watched_date}:${r.slug}`,
        content: `watched ${r.name} (${r.year})${star ? ` — ${star}` : ''}${rewatchMarker}`,
        ts: r.watched_date ? new Date(r.watched_date).toISOString() : new Date().toISOString(),
        meta: { kind: 'letterboxd_diary', name: r.name, year: r.year, rating: r.rating,
                rewatch: r.rewatch, tags: r.tags, watched_date: r.watched_date, uri: r.uri },
      });
    }
    try { renameSync(full, join(processed, f)); } catch (e) { ctx.log(`letterboxd: move failed: ${e.message}`); }
  }

  if (events.length > 0) await ctx.capture(events);
  return { count: events.length };
}
